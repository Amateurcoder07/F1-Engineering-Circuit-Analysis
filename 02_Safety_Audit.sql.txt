create view v_F1_Safety Audit as
select
c.name as CircuitName,
c.circuit_type,
c.elevation_change_m,
count(res.resultId) as TotalStarts,
sum(case when s.status in ('Accident', 'Collision', 'Spun off') then 1 else end) as InfrastructureIncidents, 
(cast(sum(case when s.status in ('Accident', 'Collision', 'Spun off') then 1 else 0 end) as float) / count(res.resultId)) * 100 as IncidentRate
from results res
join status s ON res.statusId = s.statusId
join races ra ON res.raceId = ra.raceId
join circuits c ON ra.circuitId = c.circuitId
group by c.name, c.circuit_type, c.corner_count, c.elevation_change_m;
go
